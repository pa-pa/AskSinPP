63  // send the brightness every 5 minutes to the master
64  #define LIGHTCYCLE seconds2ticks(5*60)

77  // return timer ticks
78  uint32_t getMinInterval () {
79  switch( this->getList1().minInterval() ) {
80  case 0: return seconds2ticks(15);
81  case 1: return seconds2ticks(30);

139  // this runs synch to application
140  virtual void trigger (__attribute__ ((unused)) AlarmClock& clock) {
141  if( quiet.enabled == false ) {
142  // reset state timer because motion will be send now
143  sysclock.cancel(cycle);
144  cycle.set(LIGHTCYCLE);
145  sysclock.add(cycle);
146 
147  DPRINTLN(F("Motion"));
148  // start timer to end quiet interval
149  quiet.tick = getMinInterval();
150  quiet.enabled = true;
151  sysclock.add(quiet);
152  // blink led
153  if( this->device().led().active() == false ) {
154  this->device().led().ledOn( centis2ticks(this->getList1().ledOntime()) / 2);
155  }
156  MotionEventMsg& msg = (MotionEventMsg&)this->device().message();
157  msg.init(this->device().nextcount(),this->number(),++counter,status(),this->getList1().minInterval());
158 
159  pirInterruptOff();
160  this->device().sendPeerEvent(msg,*this);
161  pirInterruptOn();
162  }
163  else if ( this->getList1().captureWithinInterval() == true ) {
164  // we have had a motion during quiet interval
165  quiet.motion = true;
166  }
167  }
168 
169  // runs in interrupt
170  void motionDetected () {
171  if( isrenabled==true ) {
172  // cancel may not needed but anyway
173  sysclock.cancel(*this);
174  // activate motion message handler
175  sysclock.add(*this);
176  }
177  }
178 
179  // we are in quiet mode - means we have a motion detected
180  bool isMotion () const {
181  return quiet.enabled == true;
182  }
